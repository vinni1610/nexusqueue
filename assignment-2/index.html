TypeScript & Python SDKs

22+ file formats. One API.

JavaScript and Python SDKs to upload your data with 22+ supported file formats. Ingest, search, and chat — all in a few lines.

import { Agentset } from "agentset";

const agentset = new Agentset({ apiKey: "agentset_xxx" });
const namespace = agentset.namespace("ns_1234");

const ingestJob = await namespace.ingestion.create({
  payload: {
    type: "FILE",
    fileUrl: "https://example.com/doc.pdf",
    fileName: "my-document.pdf"
  },
  config: { metadata: { foo: "bar" } }
});
import { Agentset } from "agentset";

const results = await new Agentset({ apiKey: "agentset_xxx" })
  .namespace("ns_1234")
  .search.query({
    query: "payment terms section 4.2",
    topK: 5,
    filters: { metadata: { type: "contract" } }
  });

console.log(results.hits);
const stream = await new Agentset({ apiKey: "agentset_xxx" })
  .namespace("ns_1234")
  .chat.stream({
    messages: [{ role: "user", content: "What are the key terms?" }],
    citations: true
  });

for await (const chunk of stream) {
  process.stdout.write(chunk.text);
}
